SArray.distinctSort ()

macro distinctSort (compfunc) : SArray

Возвращает новый массив SArray уникальных элементов исходного массива в порядке возрастания согласно функции-компаратору compfunc. Если compfunc не задана или равняется null, сравнение будет проводиться по ссылке для объектов и по значению для примитивов через > , < , ==. В среднем работает быстрее чем distinct.

+

import bankinter;
var arr = SArray(5, 3, 4, 1, 2, 5, 3);
var dist = arr.distinctSort();
dist.toScreen();  // [1, 2, 3, 4, 5]

import "../mac/extra/utils/SArray.mac";
import bankinter;
var arr = SArray(5, 3, 4, 1, 2, 5, 3);
var dist = arr.distinctSort();
dist.toScreen();  // [1, 2, 3, 4, 5]
