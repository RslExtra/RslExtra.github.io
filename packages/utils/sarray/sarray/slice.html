SArray.slice ()

macro slice (from, to) : SArray

Возвращает новый массив состоящий из элементов текущего с индекса from (включительно) по индекс to(исключительно). Если не указан или равен null аргумент from - копироваться будет с 0 аргумента. Если не указан или равен null аргумент to - копироваться будет до конца массива. Так вызывая метод без параметров, будет создаваться копия исходного массива, при этом примитивы будут скопированны полностью, в случае объектов - скопируются ссылки.

Пример:

+

var arr = SArray(1, 2, 3, 12, 18);

arr.slice(0, 3).toScreen();    // [1, 2, 3]
arr.slice(3).toScreen();       // [12, 18]
arr.slice(null, 2).toScreen(); // [3, 12, 18]
arr.slice().toScreen();        // [1, 2, 3, 12, 18]

import "../mac/extra/utils/SArray.mac";
var arr = SArray(1, 2, 3, 12, 18);

arr.slice(0, 3).toScreen();    // [1, 2, 3]
arr.slice(3).toScreen();       // [12, 18]
arr.slice(null, 2).toScreen(); // [3, 12, 18]
arr.slice().toScreen();        // [1, 2, 3, 12, 18]
