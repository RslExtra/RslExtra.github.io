ListInterface.toArray ()

macro toArray () : SArray

Возвращает массив типа SArray состоящий из элементов исходного списка.

+

list.addItems(1, 2, 3, 4);
var result = list.toArray();
for (var el, result)
    print(el + " ");            // выводим все элементы массива
end;

import "..\\mac\\extra\\collections\\collections.mac";
var list = ArrayList();
list.addItems(1, 2, 3, 4);
var result = list.toArray();
for (var el, result)
    print(el + " ");            // выводим все элементы массива
end;
